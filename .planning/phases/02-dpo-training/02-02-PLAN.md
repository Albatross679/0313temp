**If DPO improves over baseline (either variant wins):**

1. Load the winning DPO model checkpoint
2. Run test inference to generate outputs at the SUBMISSION paths:

```python
import sys, os, glob
sys.path.insert(0, '.')
from part1.config import T5DPOConfig, T5DPOConfig_lora
from part1.train import test_inference, cleanup_vram
from part1.model import initialize_model, load_model_from_checkpoint
from part1.model_flightdb import FlightSQLVocab, T5ForFlightSQL
from part1.data import load_t5_data
from src.utils.gpu_lock import GpuLock
import torch

# Select the winning DPO config
# If full FT DPO won:
cfg = T5DPOConfig()
# If LoRA DPO won:
# cfg = T5DPOConfig_lora()

device = cfg.device

_, _, test_loader = load_t5_data(
    cfg.batch_size, cfg.test_batch_size,
    input_prefix=cfg.input_prefix, include_schema=cfg.include_schema,
    schema_mode=getattr(cfg, "schema_mode", "tables"),
)

# Find the best DPO checkpoint
# For full FT: output/t5_ft_dpo_*/checkpoints/model_best.pt
# For LoRA: output/t5_ft_dpo_lora_*/checkpoints/model_best.pt
ckpt_pattern = "output/t5_ft_dpo_*/checkpoints/model_best.pt"  # adjust for winner
ckpt_path = sorted(glob.glob(ckpt_pattern))[-1]
ckpt_dir = os.path.dirname(ckpt_path)

model = load_model_from_checkpoint(
    ckpt_dir, finetune=True,
    model_checkpoint=cfg.model_checkpoint, dropout=cfg.dropout,
    best=True, device=device,
)
sql_vocab = FlightSQLVocab()
sql_vocab.to(device)

# If LoRA won, apply LoRA adapters before loading state dict
# (the checkpoint contains base + LoRA weights)
# Note: load_model_from_checkpoint loads the raw T5 state dict.
# For LoRA checkpoints saved by T5ForFlightSQL.state_dict(), the peft
# key cleaning already happened, so we load into base model directly.

model = T5ForFlightSQL(model, sql_vocab)
model.eval()

with GpuLock():
    test_inference(cfg, model, test_loader,
                   "results/t5_ft_test.sql", "records/t5_ft_test.pkl", device)
cleanup_vram()
print("Test outputs regenerated from DPO model")
```
